并查集

创建日期：2016-10-23

简介

并查集（Union Find Set）用于不相交集合的查询与合并。

它只有 2 个操作：

  • 查询两个元素是否在同一集合中。

  • 合并两个元素所在集合。

使用路径压缩的并查集的平摊复杂度接近 O(1) 。

代码

struct UnionFindSet{
    int dad[maxn];
    void init(){
        for(int i=0; i<maxn; ++i) dad[i]= i;
    }
    int get_dad(int u){
        int ans= dad[u];
        while(ans!=dad[ans]) ans= dad[ans];
        return dad[u]= ans;
    }
    void merge(int a, int b){
        dad[get_dad(a)]= get_dad(b);
    }
    bool together(int a, int b){
        return get_dad(a)==get_dad(b);
    }
};

其他

程序的测试可以到这里，下面是该题的完整代码： 
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
 
const int maxn= 10005;
 
struct UnionFindSet{
    int dad[maxn];
 
    void init(){
        for(int i=0; i<maxn; ++i) dad[i]= i;
    }
 
    int get_dad(int u){
        int ans= dad[u];
        while(ans!=dad[ans]) ans= dad[ans];
        return dad[u]= ans;
    }
 
    void merge(int a, int b){
        dad[get_dad(a)]= get_dad(b);
    }
 
    bool together(int a, int b){
        return get_dad(a)==get_dad(b);
    }
 
}ufs;
 
int main(){
    int e_size, o_size;
    scanf("%d%d", &e_size, &o_size);
 
    ufs.init();
    for(int i=0; i<o_size; ++i){
        int a, b, o;
        scanf("%d%d%d", &o, &a, &b);
        if(o==1){
            ufs.merge(a, b);
        }
        else if(o==2){
            printf("%c\n", (ufs.together(a, b)? 'Y':'N'));
        }
    }
    return 0;
}
