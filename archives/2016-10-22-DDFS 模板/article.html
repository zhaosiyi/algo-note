DDFS 模板

创建日期：2016-10-22 
vector<int> G[maxn]; // G[i][j] 表示 i 可到达的第 j 个点。
bitset<maxn> visited;
void dfs(int u){
    visited.set(u);
    previsit(u);
    for(int i=0, end= G[i].size(); i<end; ++i){
        int now= G[u][i];
        if(!visited(now)) dfs(now);
    }
    postvisited(u);
}

其中 maxn 表示点的个数。

previsit(u) 表示遍历 u 的相邻点之前的操作。

postvisit(u) 表示遍历 u 的相邻点之后的操作。

End.
