Kruskal 算法

创建日期：2016-10-23

简介

Kruskal 算法用于解决无向图的最小生成树问题（Minimal Spanning Tree，MST）。

该算法利用贪心策略，其时间复杂度为 O（边数）。

该算法能在求出最小生成树的同时判断图是否连通。

如果算法结束后最小生成树的边数等于点数-1，则图连通；小于则不连通。

算法需要用到并查集来判断与合并连通分量。

伪代码

将边按权值升序排序。
令每个点自成一个独立的连通分量。
for（e=权值从小到大的每一条边）{
    if（e所连两点不在同一连通分量中）{
        将 e 加入最小生成树。
        合并 e 所连两点所在的连通分量。
    }
}

C++ 代码

UnionFindSet ufs;
vector<Edge> edges, ans;
void kruskal(){
    sort(edges.begin(), edges.end());
    ufs.init();
    for(int i=0, end=edges.size(); i<end; ++i){
        Edge &e= edges[i];
        if(!ufs.together(e.from, e.to)){
            ans.push_back(e);
            ufs.merge(e.from, e.to);
        }
    }
}

#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
const int maxn= 5005;
struct Edge{
    int from, to, dist;
    bool operator<(const Edge &e)const{
        return dist<e.dist;
    }
};
struct UnionFindSet{
    int dad[maxn];
    void init(){
        for(int i=0; i<maxn; ++i) dad[i]= i;
    }
    int get_dad(int u){
        int ans= dad[u];
        while(ans!=dad[ans]) ans= dad[ans];
        return dad[u]= ans;
    }
    void merge(int a, int b){
        dad[get_dad(a)]= get_dad(b);
    }
    bool together(int a, int b){
        return get_dad(a)==get_dad(b);
    }
};
UnionFindSet ufs;
vector<Edge> edges, ans;
void kruskal(){
    sort(edges.begin(), edges.end());
    ufs.init();
    for(int i=0, end=edges.size(); i<end; ++i){
        Edge &e= edges[i];
        if(!ufs.together(e.from, e.to)){
            ans.push_back(e);
            ufs.merge(e.from, e.to);
        }
    }
}
int main(){
    int n_size, e_size;
    scanf("%d%d", &n_size, &e_size);
    for(int i=0; i<e_size; ++i){
        int from, to, dist;
        scanf("%d%d%d", &from, &to, &dist);
        edges.push_back((Edge){from, to, dist});
    }
    
    kruskal();
    if(ans.size()+1==n_size){
        int sum=0;
        for(int i=0, end=ans.size(); i<end; ++i){
            sum+= ans[i].dist;
        }
        printf("%d\n", sum);
    }
    else printf("orz\n");
    return 0;
}
