DFS 模板

创建日期：2016-10-22 
// vector 在头文件 <vector> 中。
// bitset 在头文件 <bitset> 中。
// maxn 表示点的个数。
vector<int> G[maxn]; // G[i][j] 表示 i 可到达的第 j 个点。
bitset<maxn> visited;
void dfs(int u){
    visited.set(u);
    previsit(u); // 遍历 u 的相邻点之前的操作。
    for(int i=0, end= G[i].size(); i<end; ++i){
        int now= G[u][i];
        if(!visited(now)) dfs(now);
    }
    postvisited(u); // 遍历 u 的相邻点之后的操作。
}
